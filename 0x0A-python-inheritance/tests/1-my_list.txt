# Import the MyList class
>>> MyList = __import__('1-my_list').MyList

# Create a MyList instance and populate it with integers
>>> my_list = MyList()

>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)

>>> print(my_list)

>>> my_list.print_sorted()  # This should print [1, 2, 3, 4, 5]

>>> print(my_list)
